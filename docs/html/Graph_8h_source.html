1009inline bool Graph::removeEdge(int sourceId, int destId) const {
1010 Vertex* source = findVertex(sourceId);
1011 Vertex* dest = findVertex(destId);
1012
1013 if (source == nullptr || dest == nullptr) {
1014 return false; // Source or destination vertex not found
1015 }
1016
1017 for (auto it = source->getAdj().begin(); it != source->getAdj().end(); ++it) {
1018 Edge * edge = *it;
1019
1020 if (edge->getSource()->getId() == sourceId && edge->getDestination()->getId() == destId) {

1023 Edge * e = *it2;
1024 if(e->getId() == edge->getId()){

1069inline bool Graph::addBidirectionalEdge(int sourceId, int destId, int edgeId , double weight) {
1070
1071 Vertex* source = findVertex(sourceId);
1072 Vertex* dest = findVertex(destId);
1073 if (source == nullptr || dest == nullptr) {
1074 return false; // Source or destination vertex not found
1075 }
1076
1077 Edge* edge = new Edge(source, dest, edgeId, weight);
1078 edgeId++;
1079 Edge* reveresEdge = new Edge(dest , source , edgeId , weight);
1080
1081 source->addAdjEdge(edgeId , edge);
1082 dest->addIncEdge(edgeId , edge);
1083
1084 source->addIncEdge(edgeId , reveresEdge);
1085 dest->addAdjEdge(edgeId , reveresEdge);
1086
1087 edgeSet[edge->getId()] = edge;
1088 edgeSet[reveresEdge->getId()] = reveresEdge;
1089
1090 return true;

1189inline bool Graph::makeFullyConnected() {
1190 // Get all vertices and edges
1191 std::unordered_map<int, Vertex*> vertices = getVertexSet();
1192 std::unordered_map<int, Edge*> edges = getEdgeSet();
1193
1194 // Check if the graph is already fully connected
1195 int numVertices = getNumVertex();
1196 int expectedEdges = numVertices * (numVertices - 1);
1197 if (edges.size() == expectedEdges || edges.size() == expectedEdges / 2) {
1198 return false;
1199 }
1200
1201 // Check if all vertices have defined coordinates
1202 for (const auto& v : vertices) {
1203 if (v.second->getLatitude() == DBL_MAX || v.second->getLongitude() == DBL_MAX) {
1204 return false;
1205 }
1206 }
1207
1208 // Create a set to store unique pairs of connected vertices
1209 std::vector<std::vector<bool>> connections(numVertices, std::vector<bool>(numVertices, false));
1210 for (const auto& e : edges) {
1211 int sourceId = e.second->getSource()->getId();
1212 int destId = e.second->getDestination()->getId();
1213 connections[sourceId][destId] = true;
1214 connections[destId][sourceId] = true; // Because of bidirectional edges
1215 }
1216
1217 // Iterate through all pairs of vertices to add missing edges
1218 for (auto it1 = vertices.begin(); it1 != vertices.end(); ++it1) {
1219 for (auto it2 = std::next(it1); it2 != vertices.end(); ++it2) {
1220 int sourceId = it1->second->getId();
1221 int destId = it2->second->getId();
1222
1223 // If there is no edge between sourceId and destId, add it
1224 if(!connections[sourceId][destId]){
1225 Vertex* v1 = it1->second;
1226 Vertex* v2 = it2->second;
1227 double weight = Harverstein(v1->getLongitude(), v1->getLatitude(), v2->getLongitude(), v2->getLatitude());
1228 addBidirectionalEdge(sourceId, destId, edgeSet.size() + 1, weight);
1229 connections[sourceId][destId] = true;
1230 connections[destId][sourceId] = true;
1231 }
1232 }
1233 }
1234 return true;
1235}

1246inline double Graph::Harverstein(double longitude1, double latitude1, double longitude2, double latitude2) const{
1247 double lon1_rad = longitude1 * M_PI / 180.0;
1248 double lat1_rad = latitude1 * M_PI / 180.0;
1249 double lon2_rad = longitude2 * M_PI / 180.0;
1250 double lat2_rad = latitude2 * M_PI / 180.0;
1251
1252 double dlon = lon2_rad - lon1_rad;
1253 double dlat = lat2_rad - lat1_rad;
1254 double a = pow(sin(dlat / 2), 2) + cos(lat1_rad) * cos(lat2_rad) * pow(sin(dlon / 2), 2);
1255 double c = 2 * atan2(sqrt(a), sqrt(1 - a));
1256
1257 const double earth_radius_km = 6371000.0;
1258
1259 double distance = earth_radius_km * c;
1260 return distance;
1261}

1300inline void Graph::orderVertexAdj(Vertex * v){
